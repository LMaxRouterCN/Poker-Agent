// 原理：从 Vuex store 的 state.Conversation.dataList 直接读取原始 markdown，
// 然后替换页面上对应的 .answer-content-wrap 元素内容。
// 不需要拦截任何网络请求，数据已经在内存里了。
// 使用 MutationObserver 事件链替代 setTimeout，确保替换时机精确。
(function() {
    'use strict';

    var TAG = '[RawMD]';

    // ========== 工具函数 ==========

    // HTML 转义，防止原始 markdown 中的 < > & 被浏览器当成 HTML 解析
    function escapeHtml(text) {
        var div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }

    // 从单条消息中提取 content_type === 'text' 的原始 markdown
    // answerArray 是多步数组，包含 think/tool_calls/tool_result/text 等类型
    // 我们只需要 text 类型，那就是 LLM 输出的原始 markdown
    function getRawMarkdown(message) {
        if (!message || !Array.isArray(message.answerArray)) return null;
        var parts = [];
        for (var i = 0; i < message.answerArray.length; i++) {
            var item = message.answerArray[i];
            // 只取最终回答，跳过思考过程和工具调用
            if (item.content_type === 'text' && typeof item.text === 'string' && item.text.length > 0) {
                parts.push(item.text);
            }
        }
        return parts.length > 0 ? parts.join('\n\n') : null;
    }

    // 将原始 markdown 文本替换到指定的 DOM 元素中
    // 用 <pre> 包裹，等宽字体，暗色背景，无圆角，左对齐
    function replaceWithRaw(wrapEl, rawText) {
        if (!wrapEl || !rawText) return;
        wrapEl.innerHTML = '<pre style="'
            + 'white-space:pre-wrap;'
            + 'word-wrap:break-word;'
            + "font-family:Consolas,Monaco,'Courier New',monospace;"
            + 'margin:0;'
            + 'padding:16px;'
            + 'background:#1a1a2e;'
            + 'color:#e0e0e0;'
            + 'border-radius:0;'
            + 'font-size:13.5px;'
            + 'line-height:1.7;'
            + 'overflow-x:auto;'
            + 'tab-size:4;'
            + 'text-align:left;'
            + '">'
            + escapeHtml(rawText)
            + '</pre>';
    }

    // 从 Vue 3 app 获取 Vuex store
    // 已验证路径：app.config.globalProperties.$store（v2 探测脚本确认可用）
    function getStore() {
        var appEl = document.getElementById('app');
        if (!appEl || !appEl.__vue_app__) return null;
        var app = appEl.__vue_app__;
        // 主路径（v2 验证过）
        if (app.config && app.config.globalProperties && app.config.globalProperties.$store) {
            return app.config.globalProperties.$store;
        }
        // 备用路径
        if (app._instance && app._instance.proxy && app._instance.proxy.$store) {
            return app._instance.proxy.$store;
        }
        return null;

            var answerEl = wrapEl.closest('.answer');
            if (!answerEl) continue;

            var match = answerId.match(/row-answer-(\d+)/);
            if (!match) continue;
